sst_clean <- sst |> # rename
  mutate(
    year  = year(date),
    month = month(date)) |>  # create new columns
      
    filter(year >= 2018)|>  # filter to match structure
  group_by(year, month) |> # group by new columns 
  summarize(
    mean_monthly_sst = mean(temp), # add in mean

  mutate(
